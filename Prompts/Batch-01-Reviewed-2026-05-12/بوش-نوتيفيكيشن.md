A. Technology Decision: FCM vs Web Push (VAPID)

  ✅ Recommended: Firebase Cloud Messaging (FCM)
  
  السبب:
    - Firebase infra جاهزة (auth, firestore, storage)
    - iOS Safari support أنظف مع FCM
    - Token-based (أبسط من VAPID subscription endpoints)
    - Free tier سخي (unlimited messages)
    - SDK جاهز (`firebase/messaging`)
    - Automatic retry + delivery receipts
  
  Trade-off accepted:
    - Vendor lock-in to Firebase (مش مشكلة — الـ stack بالفعل Firebase)
  
  ⚠️ بس: للـ iOS Safari، FCM يـ require الـ user يـ install الـ PWA
     على الـ home screen أولاً (Add to Home Screen). ده لازم UX
     guidance واضح.

C. Service Worker Extensions (public/sw.js)

  أضف في الـ existing sw.js:
  
  /* Push event handler */
  self.addEventListener('push', (event) => {
    if (!event.data) return;
    
    let payload;
    try {
      payload = event.data.json();
    } catch (e) {
      payload = { title: 'CLARK', body: event.data.text() };
    }
    
    const { title, body, icon, badge, image, tag, data, actions, urgency } = payload;
    
    const options = {
      body,
      icon: icon || '/icon-192.png',
      badge: badge || '/icon-192.png',
      image,                            // optional banner image
      tag: tag || 'clark-' + Date.now(),// groups similar notifications
      data: data || {},                 // for click handler
      actions: actions || [],           // up to 2 action buttons
      dir: 'rtl',                       // Arabic RTL
      lang: 'ar-EG',
      vibrate: [200, 100, 200],         // mobile vibration pattern
      requireInteraction: urgency === 'high',
      silent: urgency === 'low',
      timestamp: Date.now(),
    };
    
    event.waitUntil(
      self.registration.showNotification(title || 'CLARK', options)
    );
  });
  
  /* Notification click handler */
  self.addEventListener('notificationclick', (event) => {
    event.notification.close();
    
    const data = event.notification.data || {};
    const action = event.action;
    
    /* Action-specific routing */
    let url = '/';
    if (data.url) {
      url = data.url;
    } else if (data.type === 'treasury') {
      url = '/?tab=treasury' + (data.entryId ? '&entryId=' + data.entryId : '');
    } else if (data.type === 'task') {
      url = '/?tab=tasks' + (data.taskId ? '&taskId=' + data.taskId : '');
    } else if (data.type === 'instruction') {
      url = '/?tab=home&inst=' + data.instructionId;
    } else if (data.type === 'warning') {
      url = '/?tab=' + (data.target || 'home');
    }
    
    /* Handle action buttons */
    if (action === 'approve' && data.actionUrl) {
      url = data.actionUrl + '?action=approve';
    } else if (action === 'snooze' && data.snoozeUrl) {
      url = data.snoozeUrl;
    }
    
    event.waitUntil(
      clients.matchAll({ type: 'window', includeUncontrolled: true }).then(wins => {
        /* Focus existing tab if any */
        for (const win of wins) {
          if (win.url.includes(self.location.origin)) {
            win.focus();
            win.postMessage({ type: 'NOTIFICATION_CLICK', data, action, url });
            return;
          }
        }
        /* Otherwise open new */
        return clients.openWindow(url);
      })
    );
  });
  
  /* Subscription renewal */
  self.addEventListener('pushsubscriptionchange', (event) => {
    event.waitUntil(
      self.registration.pushManager.subscribe(event.oldSubscription.options)
        .then(sub => {
          /* Send new subscription to server */
          fetch('/api/notifications/renew-subscription', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ subscription: sub.toJSON() })
          });
        })
    );
  });

D. Client-side Setup (src/utils/notifications.js — جديد)

  import { getMessaging, getToken, onMessage } from "firebase/messaging";
  import { app, db } from "../firebase";
  
  let messaging = null;
  
  export async function initNotifications(user) {
    if (!('Notification' in window) || !('serviceWorker' in navigator)) {
      return { supported: false };
    }
    
    /* iOS Safari requires PWA install check */
    const isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent);
    const isStandalone = window.matchMedia('(display-mode: standalone)').matches
                      || window.navigator.standalone;
    
    if (isIOS && !isStandalone) {
      return { 
        supported: true, 
        requiresInstall: true,
        instructions: 'لتفعيل الإشعارات على iOS، يجب تثبيت التطبيق على الشاشة الرئيسية أولاً'
      };
    }
    
    try {
      messaging = getMessaging(app);
      
      /* Foreground message handler */
      onMessage(messaging, (payload) => {
        /* Show toast or in-app notification while app is open */
        window.dispatchEvent(new CustomEvent('clark-push-foreground', { detail: payload }));
      });
      
      return { supported: true, ready: true };
    } catch (e) {
      console.error('FCM init failed:', e);
      return { supported: false, error: e.message };
    }
  }
  
  export async function requestPermissionAndSubscribe(user) {
    const permission = await Notification.requestPermission();
    if (permission !== 'granted') {
      return { ok: false, reason: 'permission_denied' };
    }
    
    if (!messaging) return { ok: false, reason: 'messaging_not_initialized' };
    
    const vapidKey = import.meta.env.VITE_FIREBASE_VAPID_KEY;
    const token = await getToken(messaging, { vapidKey });
    
    if (!token) return { ok: false, reason: 'token_failed' };
    
    /* Send to server to save subscription */
    const idToken = await user.getIdToken();
    const response = await fetch('/api/notifications/subscribe', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${idToken}`,
      },
      body: JSON.stringify({
        fcmToken: token,
        device: detectDevice(),
      }),
    });
    
    return { ok: true, token };
  }
  
  function detectDevice() {
    const ua = navigator.userAgent;
    const isMobile = /Android|iPhone|iPad/i.test(ua);
    const isStandalone = window.matchMedia('(display-mode: standalone)').matches;
    
    let os = 'unknown';
    if (/iPhone|iPad|iPod/.test(ua)) os = 'ios';
    else if (/Android/.test(ua)) os = 'android';
    else if (/Win/.test(ua)) os = 'windows';
    else if (/Mac/.test(ua)) os = 'macos';
    else if (/Linux/.test(ua)) os = 'linux';
    
    let browser = 'unknown';
    if (/Chrome/.test(ua) && !/Edge/.test(ua)) browser = 'chrome';
    else if (/Safari/.test(ua) && !/Chrome/.test(ua)) browser = 'safari';
    else if (/Firefox/.test(ua)) browser = 'firefox';
    else if (/Edge/.test(ua)) browser = 'edge';
    
    return {
      type: isMobile ? (isStandalone ? 'mobile_pwa' : 'browser') : 'desktop_pwa',
      os, browser, userAgent: ua,
    };
  }

## 💡 ملاحظات لـ Ahmed قبل ما تستخدم الـ برومبت

**ده فيتشر متوسط-كبير (4-5 أسابيع شغل) بـ ROI عالي جداً.** الـ PWA infrastructure ممتازة بالفعل (V21.9.21 SW متطور) — يعني الـ foundation 50% جاهز.

**القرار المعماري الحرج:**

اخترت **Firebase Cloud Messaging (FCM)** بدلاً من Web Push Protocol المباشر:
- الـ Firebase infra جاهز بالفعل
- iOS Safari support أنظف
- Token-based أبسط من VAPID endpoints
- Free + unlimited messages
- Automatic retry + delivery receipts

**5 أسئلة الـ AI هيسألك عنها — جهز إجاباتك:**

1. **Default channels:** push only، أم push + WhatsApp + email؟
   - **توصيتي:** push default، WhatsApp opt-in للأدمن (critical warnings فقط)

2. **Critical warnings تتجاوز quiet hours؟**
   - **توصيتي:** نعم — لو حصل حاجة خطيرة (data corruption, security event)، لازم يصل حتى لو نايم

3. **Default subscriptions حسب الـ role:**
   - **توصيتي:**
     - admin: كل الـ categories
     - manager: treasury + tasks + warnings
     - sales_accountant: tasks + instructions + warnings
     - viewer: tasks فقط (assigned-to-me)

4. **Sound preferences:** silent default، أم vibration؟
   - **توصيتي:** vibration default، sound opt-in (لتجنب الإزعاج)

5. **In-app notifications:** تكمّل الـ existing toast، أم تستبدلها؟
   - **توصيتي:** تكمّلها — toast للـ confirmations في الـ app، push للـ events لما الـ app مقفول

**أكتر 6 features احترافية اقترحتها:**

1. **Quiet Hours** — DND configurable، critical warnings فقط تتجاوزها
2. **Multiple Devices** — iPhone + Mac + iPad subscriptions منفصلة (revoke individual)
3. **Rich Actions** — 2 action buttons (موافقة/رفض، تأجيل/تم)
4. **Smart Batching** — 50 notifications/يوم تتحول لـ 3 batched (morning/midday/evening)
5. **WhatsApp Fallback** — لو الـ push فشل لـ 24h+، يتحول لـ Bridge
6. **Anomaly Detection** — alert لو patterns غريبة (15 حذف في ساعة = abnormal)

**أكتر feature مبتكرة:**
**AI-Generated Daily Summary** — الـ AI Agent (لو #5 implemented) بـ يولّد ملخص يومي شخصي للأدمن: "اليوم: بعت X، حصلت Y، الأنشطة عادية. تحذير: فاتورة INV-Z لسه ما اتـ confirm-ت."

**نقطة technical حرجة — iOS Safari:**
iOS 16.4+ يدعم Web Push **بس** بعد PWA install (Add to Home Screen). لو الـ user على Safari iOS مش installed، الـ permission request هـ يفشل. الـ AI لازم:
1. Detect `isIOS && !isStandalone`
2. Show modal بـ install instructions
3. Block permission request لحد ما الـ user يـ install

**نقطة technical حرجة — VAPID/FCM Setup:**
لازم Ahmed يعمل:
1. Firebase Console → Project Settings → Cloud Messaging → Generate Web Push certificate
2. Public key → `VITE_FIREBASE_VAPID_KEY` في Vercel env vars
3. Server key → `FCM_SERVER_KEY` في Vercel env vars (أو استخدم الـ existing `FIREBASE_ADMIN_CREDENTIALS`)

**Privacy warning:**
الـ push payload visible في الـ OS notification center حتى لو الـ phone مقفول. **ممنوع** نـ embed sensitive data (full balances, customer phone numbers). الـ payload يكون "💰 تحويل خزنة جديد - اضغط للتفاصيل" والتفاصيل تظهر لما الـ user يفتح الـ app.
