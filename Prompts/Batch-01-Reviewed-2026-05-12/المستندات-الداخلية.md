C. storage.rules Addition

  match /documents/{allPaths=**} {
    allow read:   if isAuthed() && hasDocumentsReadAccess();
    allow create: if isAuthed() && hasDocumentsWriteAccess()
                  && request.resource.size < 100 * 1024 * 1024
                  && allowedDocumentMime();
    allow update: if isAuthed() && hasDocumentsWriteAccess();
    allow delete: if isManagerPlus();
  }
  
  function allowedDocumentMime() {
    return request.resource.contentType.matches('application/pdf')
        || request.resource.contentType.matches('image/.*')
        || request.resource.contentType.matches('application/.*officedocument.*')
        || request.resource.contentType.matches('application/.*ms-?excel.*')
        || request.resource.contentType.matches('application/zip')
        || request.resource.contentType.matches('text/.*');
  }
  
  function hasDocumentsReadAccess() {
    return getRole() in ['admin','manager','sales_accountant',
                          'purchase_accountant','warehouse_keeper',
                          'payroll_accountant','viewer'];
  }
  
  function hasDocumentsWriteAccess() {
    return getRole() in ['admin','manager','sales_accountant',
                          'purchase_accountant','warehouse_keeper',
                          'payroll_accountant'];
  }

F. Drag-and-Drop Upload Logic

  ```js
  const handleDrop = async (e) => {
    e.preventDefault();
    const files = Array.from(e.dataTransfer.files);
    if(files.length === 0) return;
    
    const oversized = files.filter(f => f.size > 100 * 1024 * 1024);
    if(oversized.length){
      tell("ملفات كبيرة", `${oversized.length} ملف أكبر من 100 MB`, {danger:true});
      return;
    }
    
    setUploading(true);
    setUploadProgress({total: files.length, done: 0});
    
    for(const file of files){
      try {
        const fileId = gid();
        const sanitized = sanitizeFilename(file.name);
        const path = `documents/${currentFolderId}/${fileId}_${sanitized}`;
        const ref = storageRef(storage, path);
        
        const snapshot = await uploadBytes(ref, file, { contentType: file.type });
        const downloadURL = await getDownloadURL(snapshot.ref);
        
        let thumbnailPath = null;
        if(file.type.startsWith("image/")){
          thumbnailPath = await generateImageThumbnail(file, fileId);
        } else if(file.type === "application/pdf"){
          thumbnailPath = await generatePdfThumbnail(file, fileId);
        }
        
        await upConfig(d => {
          if(!d.documentsTree) d.documentsTree = {folders:[], files:[]};
          d.documentsTree.files.push({
            id: fileId, name: file.name, folderId: currentFolderId,
            storagePath: path, downloadURL, contentType: file.type,
            size: file.size, uploadedBy: user.email, uploadedAt: nowISO(),
            thumbnailPath, previewable: isPreviewable(file.type),
            version: 1, versions: [], accessLog: [],
          });
        });
        
        setUploadProgress(p => ({...p, done: p.done + 1}));
      } catch(e){
        showToast(`فشل رفع ${file.name}: ${friendlyStorageError(e)}`);
      }
    }
    
    setUploading(false);
    showToast(`✅ تم رفع ${files.length} ملف`);
  };
  ```

**أكتر 6 features احترافية اقترحتها:**

**نقطة technical حرجة — Storage Rules deployment:**
V21.9.21 GitHub Actions auto-deploys storage.rules. تـ edit + push → 60 ثانية → live. لو الـ AI نسى الـ deployment، uploads هتفشل بـ `storage/unauthorized`. الـ `friendlyStorageError()` الموجود بـ يـ guide الـ user للحل.

**نقطة UX حرجة — Filename Arabic:**
Firebase Storage يدعم Arabic بس الـ encoding ممكن يـ break في الـ URLs. الـ sanitize function لازم يـ keep الـ Arabic + encode الـ URL بـ `encodeURIComponent`.
