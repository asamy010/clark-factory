C. Permission Check Function

  /* src/utils/treasuryPermissions.js — جديد */
  
  export function canPerformTreasuryAction(action, entry, user, data) {
    const role = getUserRole(user, data);
    const perms = data.treasuryPermissions?.roles?.[role] || {};
    
    /* Admin bypass — always allowed */
    if(canBypassLock(user, data)) return { ok: true };
    
    /* Check per-user override */
    const override = data.treasuryPermissions?.userOverrides?.[user?.email]?.[action];
    if(override?.granted && (!override.until || today() <= override.until)){
      return { ok: true, viaOverride: true };
    }
    
    /* Check base permission */
    if(!perms[action]) return { ok: false, reason: "permission_denied" };
    
    /* Date-based checks for past actions */
    if(action === "editToday" || action === "editPast" ||
       action === "delete"    || action === "deletePast") {
      if(entry?.date) {
        const isPast = entry.date < todayISO();
        if(isPast && !perms.editPast && !perms.deletePast) {
          return { ok: false, reason: "past_date_locked" };
        }
        if(isTreasuryDateLocked(entry, data) && !perms.bypassLock){
          return { ok: false, reason: "scope_locked", lock: getLockForEntry(entry, data) };
        }
      }
    }
    
    return { ok: true };
  }

  /* Treasury-specific date lock check (different from journal lock) */
  export function isTreasuryDateLocked(entry, data) {
    const locks = data.treasuryLocks || [];
    return locks.some(lock => {
      if(lock.reversedAt) return false;
      if(lock.date !== entry.date) return false;
      return matchesScope(entry, lock);
    });
  }
  
  function matchesScope(entry, lock){
    if(lock.scope === "all") return true;
    if(lock.scope === "accounts") return lock.scopeIds.includes(entry.accountId);
    if(lock.scope === "types"){
      const acct = findTreasuryAccount(entry.accountId);
      return lock.scopeIds.includes(acct?.type);
    }
    if(lock.scope === "group") {
      const group = data.treasuryGroups?.find(g => g.id === lock.scopeIds[0]);
      return group?.accountIds?.includes(entry.accountId);
    }
    return false;
  }

## 💡 ملاحظات لـ Ahmed قبل ما تستخدم الـ برومبت

**ده فيتشر متوسط-كبير (3-4 أسابيع شغل) لكن قيمته هائلة:**
- بعده مفيش "تعديلات خفية" للمحاسب على تواريخ سابقة
- الأدمن بيـ get تقرير يومي كامل عن الخزنة بدون ما يضطر يفتح التطبيق
- في override system مرن — لو المحاسب فعلاً محتاج يعدّل، الأدمن بـ grant permission مؤقتة بـ TTL

**6 أسئلة الـ AI هيسألك عنها — جهز إجاباتك:**

1. **role جديد "treasury_accountant" منفصل** ولا نستخدم `sales_accountant` الموجود؟
   - **توصيتي:** Role جديد منفصل — أوضح وأنظف، وبيدي flexibility لو حبيت تفصل المحاسبين

2. **Notification channels** — admin بيستلم على إيه افتراضياً؟
   - **توصيتي:** in-app default، WhatsApp opt-in (لأنه يتطلب الـ Bridge يكون شغّال)

3. **Treasury groups** — admin يـ define مرة واحدة في Settings ولا runtime في الـ Close Modal؟
   - **توصيتي:** Settings — أنظف وأقل ضغط على المحاسب وقت الإقفال

4. **Admin override TTL default** — كام يوم؟
   - **توصيتي:** 7 أيام — كافي لأي correction عادي بدون ما يكون مفتوح بشكل خطير

5. **Pending unposted entries (draft) في الـ scope وقت الإقفال** — block ولا allow؟
   - **توصيتي:** Block + warning واضح — المحاسب لازم يـ post أو يحذف الـ drafts قبل الـ close

6. **الـ auto-posting من المبيعات/المشتريات في الـ scope المقفول** — يـ block برضه؟
   - **توصيتي:** Yes — consistency مهم. لو فاتورة بتاريخ مقفول، الـ system يـ refuse الـ posting ويـ alert الأدمن

**نقطة حرجة — الـ Totals Snapshot:**
وقت الإقفال، الـ totals الـ {opening, closing, in, out, perAccount} بيتـ snapshot في الـ lock entry. **بعدها**، حتى لو الـ admin override وعدّل entry قديم، الـ snapshot القديم يفضل (audit). الـ reports اللي اتبعتت يفضلوا يـ reflect الإجماليات وقت الإقفال — مش الحالة الحالية. ده عشان لو الأدمن قال "الخزنة كانت 50K يوم 5/10"، فالـ proof يفضل ثابت.

**Two-layer locking model:**
الـ feature ده بيـ build على `periodLock.js` لكن منفصل:
- `data.lockedDays[]` — للـ journal entries (موجود)
- `data.treasuryLocks[]` — للـ treasury entries (جديد)

يعني المحاسب يقدر يقفل الخزنة لـ 5/10 لكن الـ journal entries لـ 5/10 لسه مفتوحة. الـ accountant accountant يـ post الـ journals بعدين. ده pattern صحيح في الـ enterprise systems.
