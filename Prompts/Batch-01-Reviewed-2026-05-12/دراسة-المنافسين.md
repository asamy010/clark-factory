B. Store Type Detection

  src/utils/competitorScraper.js (new):
  
  async function detectStoreType(url) {
    const u = new URL(url);
    const baseUrl = `${u.protocol}//${u.host}`;
    
    /* Test 1: Shopify */
    try {
      const r = await fetch(`${baseUrl}/products.json?limit=1`, {
        headers: { 'User-Agent': 'CLARK-MarketIntel/1.0' },
        signal: AbortSignal.timeout(10000),
      });
      if (r.ok) {
        const data = await r.json();
        if (data.products) return { type: "shopify", baseUrl };
      }
    } catch (e) {}
    
    /* Test 2: WooCommerce REST */
    try {
      const r = await fetch(`${baseUrl}/wp-json/wc/store/products?per_page=1`, {
        headers: { 'User-Agent': 'CLARK-MarketIntel/1.0' },
        signal: AbortSignal.timeout(10000),
      });
      if (r.ok) {
        const data = await r.json();
        if (Array.isArray(data)) return { type: "woocommerce", baseUrl };
      }
    } catch (e) {}
    
    /* Test 3: Generic HTML */
    return { type: "html", baseUrl, needsManualSetup: true };
  }

C. Shopify Scraper (الأولوية الأولى)

  Shopify عندها `/products.json` عام (250 max per page):
  
  async function fetchShopifyProducts(baseUrl) {
    const products = [];
    let page = 1;
    const maxPages = 20;
    
    while (page <= maxPages) {
      await sleep(1000); // 1 req/sec
      
      const r = await fetch(
        `${baseUrl}/products.json?limit=250&page=${page}`,
        {
          headers: {
            'User-Agent': 'CLARK-MarketIntel/1.0 (research@clark-factory.com)',
            'Accept': 'application/json',
          },
          signal: AbortSignal.timeout(30000),
        }
      );
      
      if (!r.ok) {
        if (r.status === 429) { await sleep(60000); continue; }
        break;
      }
      
      const data = await r.json();
      if (!data.products?.length) break;
      
      products.push(...data.products);
      if (data.products.length < 250) break;
      page++;
    }
    
    return products.map(normalizeShopifyProduct);
  }
  
  function normalizeShopifyProduct(p) {
    const variants = (p.variants || []).map(v => ({
      id: v.id, title: v.title, sku: v.sku,
      price: parseFloat(v.price),
      compareAtPrice: v.compare_at_price ? parseFloat(v.compare_at_price) : null,
      available: v.available,
      inventoryQty: v.inventory_quantity,
      option1: v.option1, option2: v.option2,
    }));
    const prices = variants.map(v => v.price).filter(p => p > 0);
    
    return {
      externalId: p.id, handle: p.handle,
      url: `/products/${p.handle}`,
      title: p.title, description: stripHtml(p.body_html),
      images: (p.images || []).map(i => i.src),
      vendor: p.vendor, productType: p.product_type,
      tags: (p.tags || '').split(',').map(t => t.trim()).filter(Boolean),
      variants,
      minPrice: Math.min(...prices),
      maxPrice: Math.max(...prices),
      avgPrice: prices.reduce((a,b) => a+b, 0) / prices.length,
      hasDiscount: variants.some(v => v.compareAtPrice && v.compareAtPrice > v.price),
      discountPct: calcDiscountPct(variants),
      inStock: variants.some(v => v.available),
      createdAt: p.created_at, updatedAt: p.updated_at,
    };
  }
